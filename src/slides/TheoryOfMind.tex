\section{Theory of Mind: Erpressung der Evolution}

\begin{frame}{Evolution}
    \begin{itemize}
        \item Evolutionäre Strategie: Kontinuierliche Anpassung der Wahrscheinlichkeiten zur Kooperation, sodass eigener Score maximiert wird \cite{press-dyson-2012}
        \item Besonders einfach, wenn Strategie vom Gegner ignoriert wird: $v_p = (1, 1, 1, 1)$
        \item Theory of Mind: \textit{Geglaubte} Strategie des Gegners in eigene Strategie mit einbeziehen
        \item Was passiert, wenn ein erpressender Spieler und/oder ein evolutionärer Spieler Theory of Mind haben?
    \end{itemize}
\end{frame}

\begin{frame}{Theory of Mind}
    \begin{itemize}
        \item Erpresser gegen Evolutionären, haben beide Theory of Mind vom anderen?
        \item Beide nicht: Erfolgreiche Erpressung, kein Dilemma für Evolutionären
        \item Nur Evolutionärer: IPD wird zu Ultimatum-Game
        \begin{itemize}
            \item Erpresser macht Angebot (via Erpressungs-Faktor $\lambda$)
            \item Evolutionärer erkennt Erpressung, nimmt an oder lehnt ab
        \end{itemize}
        \item Nur Erpresser: Erfolgreiche Erpressung, kein Dilemma für Evolutionären
        \item Beide: Abhängig von Theory of Mind about Mind (Anpassung eigener Strategie aufgrund Annahme Vorhandensein Theory of Mind beim Gegner)
    \end{itemize}
\end{frame}
